#### 二维数组分组
```php
$defaultrules = array(
'id'=>1,'studioid'=>5,
'id'=>2,'studioid'=>5,
'id'=>3,'studioid'=>2,
'id'=>4,'studioid'=>3
);

$rule = [];//rule 拆分数组
foreach($defaultrules as $de){
    if(isset($rule[$de['studioid']])){
        array_push($rule[$de['studioid']],$de['id']);
    }else{
        $rule[$de['studioid']] = array($de['id']);
    }
}
$rule = array(
5=>array(1,2),
2=>array(3),
3=>array(4)
);
```
